# ~COMMANDS~ #

[-d] - libera o terminal
[-it] - deixa o terminal interativo
[-p] - usado para mapear porta
[run] - roda a imagem docker
[stop] - para o container
[start] - inicia o container
[-e] seta parametros ( AUTHOR / NAME )
[-w &PATH] seta um volume destino
[-v &PATH] seta um volume origem
[ps -a] lista todos os containers
[rm] remove um container
